58 extern void init_math_functions (struct flow *flow, unsigned long seed)
59 {
60  int rc;
61 #ifndef HAVE_LIBGSL
62  UNUSED_ARGUMENT(flow);
63 #endif /* HAVE_LIBGSL */
64  /* set rounding */
65  fesetround(FE_TONEAREST);
66 
67  /* initalize rng */
68 #ifdef HAVE_LIBGSL
69  const gsl_rng_type * T;
70  gsl_rng_env_setup();
71  T = gsl_rng_default;
72  flow->r = gsl_rng_alloc (T);
73 #endif /* HAVE_LIBGSL */
74 
75  if (!seed) {
76  /* if no seed supplied use urandom */
77  DEBUG_MSG(LOG_WARNING, "client did not supply random seed "
78  "value");
79  int data = open("/dev/urandom", O_RDONLY);
80  rc = read(data, &seed, sizeof (long) );
81  close(data);
82  if(rc == -1)
83  crit("read /dev/urandom failed");
84  }
85 
86 #ifdef HAVE_LIBGSL
87  gsl_rng_set (flow->r, seed);
88  DEBUG_MSG(LOG_WARNING, "initalized local libgsl random functions for "
89  "flow %d with seed %lu, gsl generator is: %s",
90  flow->id,seed,gsl_rng_name (flow->r));
91 #else /* HAVE_LIBGSL */
92  srand((unsigned int)seed);
93  DEBUG_MSG(LOG_WARNING, "initalized posix random functions with seed "
94  "%u", (unsigned int)seed);
95 #endif /* HAVE_LIBGSL */
96 }

150 extern double dist_normal(struct flow *flow, const double mu,
151  const double sigma_square)
152 {
153 #ifdef HAVE_LIBGSL
154  const gsl_rng * r = flow->r;
155  return gsl_ran_gaussian (r, sigma_square) + mu;
156 #else /* HAVE_LIBGSL */
157  UNUSED_ARGUMENT(flow);
158  const double x = rn_uniform_minusone_to_one();
159  return (1.0 / sqrt(2.0*M_PI*sigma_square)) *
160  exp((-pow ((x-mu),2)) / (2 * sigma_square));
161 #endif /* HAVE_LIBGSL */
162 }

191 extern double dist_pareto (struct flow *flow, const double k,
192  const double x_min)
193 {
194 #ifdef HAVE_LIBGSL
195  gsl_rng * r = flow->r;
196  return gsl_ran_pareto (r, k, x_min);
197 #else /* HAVE_LIBGSL */
198  UNUSED_ARGUMENT(flow);
199  const double x = rn_uniform();
200  if (x < x_min)
201  return 0;
202  else
203  return (k/x_min) * pow (x_min/x,k+1);
204 #endif /* HAVE_LIBGSL */
205 }
