57 void fg_pcap_init()
58 {
59 /* initalize *alldevs for later use */
60 #ifdef DEBUG
61  pcap_if_t *d;
62  char devdes[200];
63 #endif /* DEBUG */
64  if (pcap_findalldevs(&alldevs, errbuf) == -1) {
65  logging_log(LOG_WARNING,"Error in pcap_findalldevs: %s\n",
66  errbuf);
67  return;
68  }
69 #ifdef DEBUG
70  for (d = alldevs; d; d = d->next) {
71  pcap_addr_t *a;
72  snprintf(devdes, sizeof(devdes), "%s: ", d->name);
73  for (a = d->addresses; a; a = a->next) {
74  char addr[100];
75  if (!a->addr)
76  continue;
77  snprintf(addr, sizeof(addr), "a=%s",
78  fg_nameinfo(a->addr, sizeof(struct sockaddr)));
79  strncat(devdes, addr, sizeof(devdes) - strlen(devdes) - 1);
80  if (a->next)
81  strncat(devdes, ", ", sizeof(devdes) - strlen(devdes) - 1);
82  }
83  DEBUG_MSG(LOG_ERR, "pcap: found pcapable device (%s)", devdes);
84  }
85 #endif /* DEBUG*/
86  pthread_mutex_init(&pcap_mutex, NULL);
87 #ifndef __DARWIN__
88  pthread_barrier_init(&pcap_barrier, NULL, 2);
89 #endif /* __DARWIN__ */
90  return;
91 }

93 void fg_pcap_cleanup(void* arg)
94 {
95  struct _flow * flow;
96  flow = (struct _flow *) arg;
97  if (!dumping)
98  return;
99  DEBUG_MSG(LOG_DEBUG, "fg_pcap_cleanup() called for flow %d", flow->id);
100  pthread_mutex_lock(&pcap_mutex);
101  if (flow->pcap_dumper)
102  pcap_dump_close((pcap_dumper_t *)flow->pcap_dumper);
103  flow->pcap_dumper = NULL;
104 
105  if (flow->pcap_handle)
106  pcap_close((pcap_t *)flow->pcap_handle);
107  flow->pcap_handle = NULL;
108  pthread_mutex_unlock(&pcap_mutex);
109  dumping = 0;

112 static void* fg_pcap_work(void* arg)
113 {
114  /* note: all the wierd casts in this function are completely useless,
115  * execpt they cirumvent strange compiler warnings because of libpcap
116  * typedef woo's */
117 
118 #ifdef DEBUG
119  struct pcap_stat p_stats;
120 #endif /* DEBUG */
121  int rc;
122  struct _flow * flow;
123  flow = (struct _flow *) arg;
124  pcap_if_t *d;
125  struct sockaddr_storage sa;
126  socklen_t sl = sizeof(sa);
127  char found = 0;
128  uint32_t net = 0;
129  uint32_t mask = 0;
130 
131  char dump_filename[500];
132  char hostname[100];
133 
134  struct bpf_program pcap_program;
135  struct timespec now;
136  char buf[60];
137 
138  DEBUG_MSG(LOG_DEBUG, "fg_pcap_thread() called for flow %d", flow->id);
139 
140  /* make sure all resources are released when finished */
141  pthread_detach(pthread_self());
142  pthread_cleanup_push(fg_pcap_cleanup, (void*) flow);
143 
144  if (getsockname(flow->fd, (struct sockaddr *)&sa, &sl) == -1) {
145  logging_log(LOG_WARNING, "getsockname() failed. Eliding "
146  "packet capture for flow.");
147  goto remove;
148  }
149 
150  /* find appropriate (used for test) interface to dump */
151  for (d = alldevs; d; d = d->next) {
152  pcap_addr_t *a;
153  for (a = d->addresses; a; a = a->next) {
154  if (!a->addr)
155  continue;
156  if (sockaddr_compare(a->addr, (struct sockaddr *)&sa)) {
157  DEBUG_MSG(LOG_NOTICE, "pcap: data connection "
158  "inbound from %s (%s)", d->name,
159  fg_nameinfo(a->addr,
160  sizeof(struct sockaddr)));
161  found = 1;
162  break;
163  }
164  }
165  if (found)
166  break;
167  }
168 
169  if (!found) {
170  logging_log(LOG_WARNING, "Failed to determine interface "
171  "for data connection. No pcap support.");
172  goto remove;
173  }
174 
175  /* Make sure errbuf contains zero-length string in order to enable
176  * pcap_open_live to report warnings. */
177  errbuf[0] = '\0';
178  flow->pcap_handle =
179  (struct pcap_t *)pcap_open_live(d->name, PCAP_SNAPLEN,
180  PCAP_PROMISC,
181  0, /* no read timeout */
182  errbuf);
183 
184  if (!flow->pcap_handle) {
185  logging_log(LOG_WARNING, "Failed to init pcap on device %s:"
186  " %s", d->name, errbuf);
187  goto remove;
188  }
189 
190 
191  if (pcap_lookupnet(d->name, &net, &mask, errbuf) < 0) {
192  logging_log(LOG_WARNING, "pcap: netmask lookup failed: %s",
193  errbuf);
194  goto remove;
195  }
196 
197  /* We rely on a non-blocking dispatch loop */
198  if (pcap_setnonblock((pcap_t *)flow->pcap_handle,
199  1 /* non-blocking */,
200  errbuf) < 0) {
201  logging_log(LOG_WARNING, "pcap: failed to set non-blocking: "
202  "%s", errbuf);
203  goto remove;
204  }
205 
206  /* compile filter */
207  if (pcap_compile((pcap_t *)flow->pcap_handle,
208  &pcap_program, PCAP_FILTER, 1, mask) < 0) {
209  logging_log(LOG_WARNING, "pcap: failed compiling filter "
210  "'%s': %s", PCAP_FILTER,
211  pcap_geterr((pcap_t *)flow->pcap_handle));
212  goto remove;
213  }
214 
215  /* attach filter to interface */
216  if (pcap_setfilter((pcap_t *)flow->pcap_handle,
217  &pcap_program) < 0) {
218  logging_log(LOG_WARNING, "pcap: failed to set filter: "
219  "%s", pcap_geterr((pcap_t *)flow->pcap_handle));
220  goto remove;
221  }
222 
223  /* generate a nice filename */
224  dump_filename[0] = '\0';
225 
226  /* dir and prefix */
227  if (dump_dir)
228  strcat(dump_filename, dump_dir);
229  if (dump_prefix)
230  strcat(dump_filename, dump_prefix);
231 
232  /* timestamp */
233  gettime(&now);
234  strftime(buf, sizeof(buf), "%Y-%m-%d-%H:%M:%S",
235  localtime(&now.tv_sec));
236  strcat(dump_filename, buf);
237 
238  /* hostname */
239  hostname[0]= '\0';
240  if (!gethostname(hostname, 59)) {
241  strcat(dump_filename, "-");
242  strcat(dump_filename, hostname);
243  }
244 
245  /* interface */
246  strcat(dump_filename, "-");
247  strcat(dump_filename, d->name);
248 
249  /* suffix */
250  strcat(dump_filename, ".pcap");
251 
252  DEBUG_MSG(LOG_NOTICE, "dumping to \"%s\"", dump_filename);
253 
254  flow->pcap_dumper = (struct pcap_dumper_t *)pcap_dump_open(
255  (pcap_t *)flow->pcap_handle, dump_filename);
256 
257  if (!flow->pcap_dumper) {
258  logging_log(LOG_WARNING, "pcap: failed to open dump file "
259  "writing: %s",
260  pcap_geterr((pcap_t *)flow->pcap_handle));
261  goto remove;
262  }
263 
264  /* barrier: dump is ready */
265 #ifndef __DARWIN__
266  pthread_barrier_wait(&pcap_barrier);
267 #endif /* __DARWIN__ */
268  for (;;) {
269  rc = pcap_dispatch((pcap_t *)flow->pcap_handle, -1,
270  &pcap_dump, (u_char *)flow->pcap_dumper);
271 
272  if (rc < 0) {
273  logging_log(LOG_WARNING, "pcap_dispatch() failed. "
274  "Packet dumping stopped for flow %d.",
275  flow->id);
276  /* cleanup automatically called */
277  pthread_exit(0);
278  }
279 #ifdef DEBUG
280  pcap_stats((pcap_t *)flow->pcap_handle, &p_stats);
281 #endif /* DEBUG */
282  DEBUG_MSG(LOG_NOTICE, "pcap: finished dumping %u packets for "
283  "flow %d", rc, flow->id);
284  DEBUG_MSG(LOG_NOTICE, "pcap: %d packets received by filter for "
285  "flow %d", p_stats.ps_recv, flow->id);
286  DEBUG_MSG(LOG_NOTICE, "pcap: %d packets dropped by kernel for "
287  "flow %d", p_stats.ps_drop, flow->id);
288  if (rc == 0)
289  /* if no packets are received try
290  * if we should cancel */
291  pthread_testcancel();
292  }
293  pthread_cleanup_pop(1);
294 
295 remove:
296 #ifndef __DARWIN__
297  pthread_barrier_wait(&pcap_barrier);
298 #endif /* __DARWIN__ */
299  return 0;
300 
301 }

303 void fg_pcap_go(struct _flow *flow)
304 {
305  int rc;
306  if (!flow->settings.traffic_dump)
307  return;
308 
309  if (dumping) {
310  logging_log(LOG_WARNING, "pcap: dumping already in progress "
311  "on this host");
312  return;
313  }
314 
315  DEBUG_MSG(LOG_DEBUG, "called fg_pcap_go() for flow %d", flow->id);
316  dumping = 1;
317  rc = pthread_create(&flow->pcap_thread, NULL, fg_pcap_work,
318  (void*)flow);
319  /* barrier: dump thread is ready (or aborted) */
320 #ifndef __DARWIN__
321  pthread_barrier_wait(&pcap_barrier);
322 #endif /* __DARWIN__ */
323  if (rc)
324  logging_log(LOG_WARNING, "Could not start pcap thread: %s",
325  strerror(errno) );
326  return;
327 }
